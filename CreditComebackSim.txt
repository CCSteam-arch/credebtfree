        let advice = 'Shorter timelines require higher monthly payments, offering a faster comeback.';
        
        if (duration === '48') {
            advice = 'Our most popular choice: a balanced approach with steady progress and manageable payments.';
        } else if (duration === '60') {
            advice = 'A comfortable pace with lower monthly commitments, ideal for budget-conscious clients.';

            advice = 'The extended plan, perfect for high balance accounts ($100,000+ total debt) with maximum flexibility.';

        else if (analysis.utilization <= 50) utilValue = 50;
        else if (analysis.utilization <= 75) utilValue = 75;
        else utilValue = 100;
        
        // Update the main React state
        updateUserData({
            ficoScore: analysis.ficoScore,
            totalDebt: analysis.totalDebt,
            accountsEnrolling: analysis.accountsEnrolling,
            positiveAccounts: analysis.positiveAccounts,
            oldestAccountAge: analysis.oldestAccountAge,
            totalCreditLimit: analysis.totalCreditLimit,

            // Keep user's preferred timeline if already set
            programTimeline: userData.programTimeline 
        });

const RecoveryTimeline = ({ results, userData }) => {
    const dipScore = results.milestoneDipScore;
    const stabilizationScore = results.milestoneStabilizationScore;
    const recoveryScore = results.milestoneRecoveryScore;
    const finalScore = results.projectedScore;
    const initialScore = results.initialScore;
    
    // Calculate point changes
    const dipChange = initialScore - dipScore;
    
    // Dynamic Milestones for the granular progress (12mo to 24mo)
    const milestones = [];
    const startMonth = 12;
    const endMonth = 24;
    const stabilizationMonth = 12; // FIX: Define stabilizationMonth locally
    const gainPerMonth = (recoveryScore - stabilizationScore) / (endMonth - startMonth);
    
    for (let m = startMonth; m <= endMonth; m += 3) { // Show progress every 3 months
         const monthsAfterStabilization = m - stabilizationMonth;
         const score = Math.round(stabilizationScore + (gainPerMonth * monthsAfterStabilization));
         
         milestones.push({
             month: m,
             score: score,
             title: `Month ${m} Progress`,
             color: 'yellow',
             description: `On track for score tier upgrade.` // 'description' instead of 'subtext'
         });
    }
    
    const timelineEvents = [
        {
            month: 0,
            score: initialScore,
            title: "Program Start",
            color: "blue",
            description: "Beginning your financial transformation."
        },
        ...(userData.scenarioType === 'pre-enrollment' ? [{
            month: "6-9",
            score: dipScore,
            title: "Strategic Dip Period",
            color: "red",
            description: `Temporary drop as negotiations begin (Expected -${dipChange} pts).`
        }] : []),
        ...milestones, // Inject granular milestones
        {
            month: userData.programTimeline,
            score: finalScore,
            title: "Program Graduation",
            color: "green",
            description: `Achieved debt freedom and excellent credit!`
        }
    ].filter(event => event.month <= userData.programTimeline || event.month === 0 || event.month === "6-9"); // Filter out milestones past graduation
     
    // De-duplicate if graduation and a milestone are the same month
     const finalEvents = timelineEvents.reduce((acc, current) => {
        const x = acc.find(item => item.month === current.month);
        if (!x) {
            return acc.concat([current]);
        } else if (current.color === 'green') { // Prioritize graduation event
             acc.pop(); // Remove the milestone
             return acc.concat([current]); // Add graduation
        } else {
            return acc; // Keep graduation
        }
    }, []);


    return (
        <div className="bg-gradient-to-br from-blue-50 to-indigo-100 rounded-xl p-6 md:p-8 border border-blue-200">
            <h3 className="text-2xl font-bold text-gray-800 mb-6 flex items-center">
                <svg className="w-6 h-6 text-blue-600 mr-3" fill="none" stroke="currentColor" viewBox="0 0 24 24">
                    <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M12 8v4l3 3m6-3a9 9 0 11-18 0 9 9 0 0118 0z"/>
                </svg>
                Your Personalized Credit Recovery Journey
            </h3>
            
            {/* Timeline Visualization */}
            <div className="relative pl-12"> {/* Increased padding for better line visibility */}
                {/* Timeline Line */}
                <div className="absolute left-10 top-0 bottom-0 w-1 bg-gradient-to-b from-red-400 via-yellow-400 to-green-400 rounded-full"></div>
                
                {/* Timeline Milestones */}
                <div className="space-y-8">
                    
                    {finalEvents.map((event, index) => {
                        const isStartingPoint = index === 0;
                        const isDip = event.color === 'red';
                        const isFinal = event.color === 'green';
                        
                        let textColor = isStartingPoint ? 'text-blue-700' : isDip ? 'text-red-700' : isFinal ? 'text-green-700' : 'text-yellow-700';
                        let bgColor = isStartingPoint ? 'bg-blue-600' : isDip ? 'bg-red-500' : isFinal ? 'bg-green-600' : 'bg-yellow-500';
                        let borderColor = isStartingPoint ? 'border-blue-600' : isDip ? 'border-red-500' : isFinal ? 'border-green-600' : 'border-yellow-500';
                        
                        return (
                            <div key={index} className="flex items-start relative">
                                {/* Timeline Dot */}
                                <div className={`w-6 h-6 ${bgColor} rounded-full flex-shrink-0 absolute left-7 top-1/2 transform -translate-x-1/2 -translate-y-1/2 shadow-lg z-10`}></div>
                                
                                {/* Event Content */}
                                <div className="ml-8 flex-1 pl-4">
                                    <div className={`rounded-xl p-4 shadow-md border-l-4 ${borderColor} ${isDip ? 'bg-red-50' : isFinal ? 'bg-green-50' : 'bg-white'}`}>
                                        <div className="flex justify-between items-start">
                                            <div>
                                                <h4 className={`font-bold ${textColor}`}>{event.title}</h4>
                                                <p className="text-gray-600 text-sm mt-1">{event.description}</p>
                                            </div>
                                            <div className="text-right flex-shrink-0 ml-4">
                                                <div className={`text-xl font-bold ${textColor}`}>{event.score}</div>
                                                <div className="text-xs text-gray-500">FICO</div>
                                            </div>
                                        </div>
                                    </div>
                                </div>
                            </div>
                        );
                    })}
                </div>
            </div>
        </div>
    );
};

    };

    // Application State
    const [currentStep, setCurrentStep] = useState(1); // Start at new Step 1
    const [userData, setUserData] = useState(defaultUserData);
    const [results, setResults] = useState(null);
    const [error, setError] = useState(null);
    const [isLoading, setIsLoading] = useState(false);
    
    // Enhanced Firebase initialization
    useEffect(() => {
        if (!firebaseConfig || !Object.keys(firebaseConfig).length) {
            console.error("Firebase configuration is missing or empty. Skipping database initialization.");
            setIsAuthReady(true);
            setUserId(crypto.randomUUID()); // Provide a fallback user ID
            return;
        }

        try {
            const app = initializeApp(firebaseConfig);
            const authInstance = getAuth(app);
            const dbInstance = getFirestore(app);

            setAuth(authInstance);
            setDb(dbInstance);

            const unsubscribe = onAuthStateChanged(authInstance, async (user) => {
                let currentUserId;
                if (!user) {
                    try {
                        if (initialAuthToken) {
                            const cred = await signInWithCustomToken(authInstance, initialAuthToken);
                            currentUserId = cred.user.uid;
                        } else {
                            const cred = await signInAnonymously(authInstance);
                            currentUserId = cred.user.uid;
                        }
                    } catch (e) {
                        console.error("Firebase Sign-in Error:", e);
                        currentUserId = crypto.randomUUID();
                    }
                } else {
                    currentUserId = user.uid;
                }
                
                setUserId(currentUserId);
                setIsAuthReady(true);
            });

            return () => unsubscribe();
        } catch (e) {
            console.error("Firebase Initialization Error:", e);
            setIsAuthReady(true);
            setUserId(crypto.randomUUID()); // Provide a fallback user ID
        }
    }, [initialAuthToken, JSON.stringify(firebaseConfig)]);

    // Enhanced utility functions
    const showError = (message) => {
        setError(message);
        setTimeout(() => setError(null), 5000);
    };

    const updateUserData = useCallback((data) => {
        setUserData(prev => ({ ...prev, ...data }));
    }, []);

    // --- CORE LOGIC (MERGED) ---

    // Enhanced weighting model (from new code)
    const calculateWeights = useCallback((ficoScore, totalDebt, monthlyIncome, positiveAccounts, oldestAccountAge, totalCreditLimit) => {
        const baseWeights = {
            paymentHistory: 0.35,
            utilization: 0.30,
            accountAge: 0.15,
            creditMix: 0.10,
            newCredit: 0.10
        };
        
        // Calculate true overall utilization stability
        const trueOverallUtilization = totalCreditLimit > 0 ? (totalDebt / totalCreditLimit) : 1; // Default to worst if limit is zero

        // Enhanced score-tier adjustments for maximum accuracy
        if (ficoScore >= 740) {
            baseWeights.paymentHistory = 0.25;
            baseWeights.utilization = 0.40; 
            baseWeights.accountAge = 0.20;
        } else if (ficoScore <= 580) {
            baseWeights.paymentHistory = 0.45; 
            baseWeights.utilization = 0.25;
            baseWeights.newCredit = 0.15; 
        }

        // Refined DTI adjustments
        const dti = totalDebt / (monthlyIncome * 12);
        if (dti > 0.45) {
            baseWeights.utilization += 0.05; 
            baseWeights.paymentHistory -= 0.05; 
        }
        
        // ENHANCED: Weighting based on Overall Utilization Stability (New Input Impact)
        if (trueOverallUtilization < 0.30) {
            // Very low overall utilization means score recovery is inherently more stable
            baseWeights.accountAge = Math.min(0.25, baseWeights.accountAge + 0.05);
            baseWeights.paymentHistory = Math.max(0.30, baseWeights.paymentHistory - 0.05);
        }

        // Enhanced positive accounts impact
        if (positiveAccounts > 0) {
            baseWeights.paymentHistory = Math.max(0.30, baseWeights.paymentHistory - 0.05);
            baseWeights.creditMix = Math.min(0.20, baseWeights.creditMix + 0.05);
        }
        
        // Refined oldest account age logic
        if (oldestAccountAge < 4) {
             baseWeights.accountAge = Math.max(0.05, baseWeights.accountAge - 0.05);
             baseWeights.newCredit = Math.min(0.20, baseWeights.newCredit + 0.05);
        } else if (oldestAccountAge > 10) {
             baseWeights.accountAge = Math.min(0.25, baseWeights.accountAge + 0.05);
        }
        
        // Normalize weights for accuracy
        const sum = Object.values(baseWeights).reduce((a, b) => a + b, 0);
        for (const key in baseWeights) {
            baseWeights[key] /= sum;
        }

        return baseWeights;
    }, []);

    // Enhanced worst-case impact calculation
    const calculateWorstCaseImpact = useCallback((data, weights) => {
        let drop = 0;
        // Use the 'utilization' from state, which is for *enrolled* accounts
        const utilizationFactor = Math.max(0, (data.utilization - 30) / 10);
        
        drop += utilizationFactor * 10 * (weights.utilization * 3);
        
        const severityFactor = Math.max(0, (700 - data.ficoScore) / 10);
        drop += severityFactor * 5;
        
        drop -= data.positiveAccounts * 5;

        if (data.oldestAccountAge > 10) {
             drop -= 10;
        }

        return Math.min(150, Math.max(20, Math.round(drop)));
    }, []);

    // Enhanced simulation engine (from new code)
    const runSimulation = useCallback((data) => {
        const initialScore = Number(data.ficoScore);
        // Pass the new totalCreditLimit to the weights function
        const weights = calculateWeights(
            initialScore, 
            Number(data.totalDebt), 
            Number(data.monthlyIncome), 
            Number(data.positiveAccounts), 
            Number(data.oldestAccountAge), 
            Number(data.totalCreditLimit)
        );
        
        let lowPointScore, projectedScore;
        let recoveryMonths = data.programTimeline;
        let impactPenalty = 0;
        
        if (data.scenarioType === 'pre-enrollment') {
            impactPenalty = calculateWorstCaseImpact(data, weights);
            lowPointScore = Math.max(300, initialScore - impactPenalty);
        } else {
            lowPointScore = initialScore;
            recoveryMonths = Math.max(12, data.programTimeline - (Number(data.monthsInProgram) || 0));
            impactPenalty = 0;
        }
        
        let totalPotentialGain = 0;
        
        totalPotentialGain += weights.utilization * 200;
        totalPotentialGain += weights.paymentHistory * 150 * (recoveryMonths / data.programTimeline);
        totalPotentialGain += weights.accountAge * 100;
        
        let toolGain = 0;
        if (data.securedCard) toolGain += 30;
        if (data.creditBuilder) toolGain += 25;
        if (data.authorizedUser) toolGain += 10;
        
        totalPotentialGain += toolGain;
        
        projectedScore = Math.min(850, lowPointScore + Math.round(totalPotentialGain));
        
        const postDebt = Number(data.totalDebt) * 0.4;
        const annualPostDebt = postDebt * 0.05 / 12; // Simplified repayment
        const dtiRatio = (annualPostDebt / Number(data.monthlyIncome)) * 100;
        
        const debtSavings = Number(data.totalDebt) * 0.6;
        
        const totalGain = projectedScore - lowPointScore;
        const stabilizationScore = lowPointScore + Math.round(totalGain * 0.1);
        const recoveryScore = lowPointScore + Math.round(totalGain * 0.65);

        return {
            initialScore,
            lowPointScore,
            projectedScore,
            scoreGain: projectedScore - initialScore,
            recoveryTime: `${data.programTimeline} months`,
            postDTI: Math.min(50, dtiRatio).toFixed(1),
            savings: debtSavings,
            impactPenalty,
            recoveryMonths,
            weights,
            milestoneDipScore: lowPointScore,
            milestoneStabilizationScore: stabilizationScore,
            milestoneRecoveryScore: recoveryScore
        };
    }, [calculateWeights, calculateWorstCaseImpact]);
    
    // Enhanced save simulation
    const saveSimulation = useCallback(async (simulationData, simulationResults) => {
        if (!db || !userId || !isAuthReady) {
            console.warn("Database or User ID not ready. Cannot save simulation.");
            return;
        }

        const simulationRef = collection(db, 'artifacts', appId, 'users', userId, 'simulations');
        
        try {
            await addDoc(simulationRef, {
                userId: userId,
                timestamp: serverTimestamp(),
                input: simulationData,
                results: simulationResults,
            });
            console.log("Simulation saved successfully!");
        } catch (e) {
            console.error("Error adding document: ", e);
            showError("Failed to save simulation results to your profile.");
        }
    }, [db, userId, appId, isAuthReady]);

    // --- Navigation ---
    
    const handlePrevious = () => {
        if (currentStep > 1) {
            setCurrentStep(currentStep - 1);
            setResults(null);
        }
    };
    
    const handleReset = () => {
        setCurrentStep(1);
        setResults(null);
        setUserData(defaultUserData);
    };

    // MERGED: Updated validation logic for the new step flow
    const handleNext = () => {
        // --- Step 1 Validation (Scenario) ---
        if (currentStep === 1) {
            if (!userData.scenarioType) {
                 showError("Please select a scenario to continue.");
                 return;
            }
            if (userData.scenarioType === 'progress-tracker') {
                const { monthsInProgram, settledAccounts } = userData;
                if (isNaN(monthsInProgram) || monthsInProgram < 0) {
                    showError("Please enter a valid number of months you've been in the program.");
                    return;
                }
                if (isNaN(settledAccounts) || settledAccounts < 0) {
                    showError("Please enter a valid number of settled accounts (0 or more).");
                    return;
                }
            }
        }

        // --- Step 2 Validation (Profile) ---
        if (currentStep === 2) {
            const { ficoScore, totalDebt, accountsEnrolling, positiveAccounts, oldestAccountAge, monthlyIncome, utilization, programTimeline, totalCreditLimit } = userData;
            
            const numFico = parseFloat(ficoScore);
            const numIncome = parseFloat(monthlyIncome);
            const numDebt = parseFloat(totalDebt);
            const numLimit = parseFloat(totalCreditLimit);
            const numAccounts = parseInt(accountsEnrolling);
            const numPositive = parseInt(positiveAccounts);
            const numAge = parseInt(oldestAccountAge);

            if (isNaN(numFico) || numFico < 300 || numFico > 850) {
                showError("Please enter a valid FICO Score between 300 and 850.");
                return;
            }
            if (isNaN(numIncome) || numIncome <= 0) {
                showError("Please enter a valid Monthly Income amount.");
                return;
            }
            if (isNaN(numDebt) || numDebt <= 0) {
                showError("Please enter a valid amount for Total Debt for Resolution.");
                return;
            }
             if (isNaN(numLimit) || numLimit < 0) {
                showError("Please enter a valid Total Credit Limit.");
                return;
            }
             if (numLimit < numDebt) {
                 showError("Your Total Credit Limit cannot be less than your Total Debt.");
                 return;
             }
            if (isNaN(numAccounts) || numAccounts < 1) {
                showError("Please enter at least 1 debt account for enrollment.");
                return;
            }
            if (isNaN(numPositive) || numPositive < 0) {
                 showError("Please enter a valid number of positive accounts (0 or more).");
                return;
            }
            if (isNaN(numAge) || numAge < 0) {
                 showError("Please enter a valid age for your oldest credit account.");
                return;
            }
            if (!utilization) {
                showError("Please select your Credit Utilization range.");
                return;
            }
            if (isNaN(programTimeline)) {
                showError("Please select an Expected Program Timeline.");
                return;
            }
        }
        
        // --- Step 3 (Submission) ---
        if (currentStep === 3) {
            calculateAndShowResults();
            return;
        }

        if (currentStep < 4) {
            setCurrentStep(currentStep + 1);
        }
    };

    const calculateAndShowResults = () => {
        setIsLoading(true);
        setResults(null); // Clear previous results
        
        if (!isAuthReady) {
            console.warn("Auth not ready, delaying simulation save...");
        }

        setTimeout(() => {
            try {
                const calculatedResults = runSimulation(userData);
                setResults(calculatedResults);
                
                saveSimulation(userData, calculatedResults);
                
                setIsLoading(false);
                setCurrentStep(4);
            } catch (simError) {
                console.error("Simulation Error:", simError);
                showError("A problem occurred during the simulation calculation.");
                setIsLoading(false);
            }
        }, 1500);
    };

    // --- Render Logic ---

    const renderStep = () => {
        if (!isAuthReady) {
             return (
                <div className="max-w-4xl mx-auto px-4">
                    <div className="bg-white rounded-2xl shadow-2xl p-8 text-center border border-gray-100">
                        <div className="animate-spin rounded-full h-12 w-12 border-b-2 border-blue-600 mx-auto mb-4"></div>
                        <p className="text-gray-700 font-semibold">Initializing Credit Hub...</p>
                        <p className="text-sm text-gray-500 mt-2">Securing your session and connecting to database.</p>
                        <p className="text-xs text-gray-400 mt-4">Session ID: {userId || 'Authenticating...'}</p>
                    </div>
                </div>
            );
        }
        
        // MERGED: New Step Order
        switch (currentStep) {
            case 1:
                return <Step1Scenario userData={userData} updateUserData={updateUserData} />;
            case 2:
                return <Step2Profile userData={userData} updateUserData={updateUserData} />;
            case 3:
                return <Step3Tools userData={userData} updateUserData={updateUserData} />;
            case 4:
                return <Step4Results results={results} isLoading={isLoading} userData={userData} onReset={handleReset} setCurrentStep={setCurrentStep} />;
            default:
                return <Step1Scenario userData={userData} updateUserData={updateUserData} />;
        }
    };
    
    // MERGED: New Step Labels
    const steps = ['Scenario', 'Profile', 'Tools', 'Results'];
    const progressWidth = currentStep === 4 ? 100 : ((currentStep - 1) / (steps.length - 1)) * 100;

    return (
        // MERGED: Added font-inter
        <div className="min-h-screen bg-gradient-to-br from-blue-600 via-purple-600 to-indigo-600 font-inter">
            {/* Sticky Header (from new code) */}
            <div className="bg-white shadow-md border-b border-gray-100 sticky top-0 z-30">
                <div className="max-w-6xl mx-auto px-4 py-4 md:py-6">
                    <h1 className="text-2xl md:text-4xl font-bold text-center text-gray-900 mb-1">Credit Comeback Simulator</h1>
                    <p className="text-base md:text-lg text-gray-600 text-center">Personalized recovery with AI accuracy</p>
                    {isAuthReady && userId && (
                        <p className="text-sm text-gray-400 text-center mt-2">
                            Session ID: <span className="font-mono text-xs bg-gray-100 px-2 py-1 rounded-md">{userId}</span>
                        </p>
                    )}
                </div>
            </div>

            <div className="p-4 md:p-10">
                {/* Error Message (from current) */}
                {error && (
                    <div className="fixed top-20 right-5 z-50 max-w-sm w-full">
                        <div className="bg-red-600 text-white p-4 rounded-xl shadow-2xl flex items-start space-x-3">
                            <span className="flex-shrink-0"><IconAlert /></span>
                            <div>
                                <h4 className="font-bold">An error occurred</h4>
                                <p className="text-sm">{error}</p>
                            </div>
                            <button onClick={() => setError(null)} className="ml-auto text-red-100 hover:text-white">&times;</button>
                        </div>
                    </div>
                )}
                
                {/* Progress Bar (from current, with new step labels) */}
                {currentStep < 4 && (
                    <div className="max-w-3xl mx-auto mb-8 bg-white/10 backdrop-blur-sm p-4 rounded-xl">
                        <h2 className="sr-only">Progress</h2>
                        <div className="relative">
                            <div className="absolute left-0 top-1/2 -translate-y-1/2 w-full h-2 bg-gray-200/50 rounded-full"></div>
                            <div 
                                className="absolute left-0 top-1/2 -translate-y-1/2 h-2 bg-green-400 rounded-full transition-all duration-500"
                                style={{ width: `${progressWidth}%` }}
                            ></div>
                            <div className="relative flex justify-between">
                                {steps.map((step, index) => (
                                    <div key={step} className="flex flex-col items-center">
                                        <div className={`w-8 h-8 rounded-full flex items-center justify-center transition-all duration-300 ${index + 1 <= currentStep ? 'bg-green-400 text-white' : 'bg-white text-gray-600'}`}>
                                            {index + 1 < currentStep ? <IconCheck /> : <span className="text-sm font-bold">{index + 1}</span>}
                                        </div>
                                        <span className={`mt-2 text-xs font-semibold ${index + 1 <= currentStep ? 'text-white' : 'text-gray-200'}`}>{step}</span>
                                    </div>
                                ))}
                            </div>
                        </div>
                    </div>
                )}

                {/* Step Content */}
                <main>
                    {renderStep()}
                </main>

                {/* Navigation (from current) */}
                {currentStep < 4 && (
                    <footer className="max-w-5xl mx-auto px-4 mt-8">
                        <div className="flex justify-between items-center">
                            <button
                                onClick={handlePrevious}
                                disabled={currentStep === 1}
                                className="px-8 py-3 bg-white text-gray-700 font-bold rounded-xl shadow-md border border-gray-200 hover:bg-gray-100 transition disabled:opacity-50 disabled:cursor-not-allowed"
                            >
                                Back
                            </button>
                            <button
                                onClick={handleNext}
                                disabled={isLoading}
                                className="px-8 py-3 bg-gradient-to-r from-green-500 to-emerald-600 text-white font-black rounded-xl shadow-lg hover:shadow-xl transition transform hover:-translate-y-0.5 flex items-center space-x-2 disabled:opacity-70"
                            >
                                {isLoading ? (
                                    <svg className="animate-spin h-5 w-5 text-white" xmlns="http://www.w3.org/2000/svg" fill="none" viewBox="0 0 24 24">
                                        <circle className="opacity-25" cx="12" cy="12" r="10" stroke="currentColor" strokeWidth="4"></circle>
                                        <path className="opacity-75" fill="currentColor" d="M4 12a8 8 0 018-8V0C5.373 0 0 5.373 0 12h4zm2 5.291A7.962 7.962 0 014 12H0c0 3.042 1.135 5.824 3 7.938l3-2.647z"></path>
                                    </svg>
                                ) : (
                                    <span>{currentStep === 3 ? 'Calculate My Results' : 'Next Step'}</span>
                                )}
                                {!isLoading && currentStep < 3 && <IconChevronRight />}
                            </button>
                        </div>
                    </footer>
                )}
            </div>
            
            {/* Trust Indicators (from new code) */}
            <TrustIndicators />
        </div>
    );
};

export default App;
